    def _read_canvas_pixel(page, x, y):
        """Read RGBA at canvas pixel (x,y). Returns [r,g,b,a]."""
        return page.evaluate(f"""(() => {{
            const canvas = document.querySelector('#coloring-canvas');
            if (!canvas) return null;
            const ctx = canvas.getContext('2d');
            const data = ctx.getImageData({x}, {y}, 1, 1).data;
            return [data[0], data[1], data[2], data[3]];
        }})()""")

    def _tap_canvas_at(page, canvas_x, canvas_y):

            canvas.dispatchEvent(new PointerEvent('pointerdown', {{

                bubbles: true, pointerId: 1, pointerType: 'touch'
