        src_dir = GENERATED_SVG_DIR / "dressup"

        dst_dir.mkdir(parents=True, exist_ok=True)
        results = []
        if not src_dir.exists():

        for svg_file in sorted(src_dir.glob("*.svg")):
            dst = dst_dir / svg_file.name

            results.append(dst)
